
class Solution {
    int[][] rotateMatrix(int k, int mat[][]) {
        // code here
        int m=mat.length;
        int n=mat[0].length;
        int temp[][] =new int[m][n];
        k=k%n;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                temp[i][j]=mat[i][(j+k)%n];
            }
            
        }
        return temp;
    }
}
